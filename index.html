What is a component?

com · po · nent

/kəmˈpəʊnənt/

reusable Vue instances with a name

Basics

Example


            <!-- counter-button -->
            <button @click="count++">
              You clicked me {{ count }} times.
            </button>
          

            {
              data: function() {
                return {
                  count: 0
                };
              }
            }

Usage


            <counter-button></counter-button>
            <counter-button></counter-button>

Registration


            Vue.component('button-counter', {
              data: function() {
                return {
                  count: 0
                };
              },
              template: `
                
              `;
            });
          

            <counter-button></counter-button>

Single File Components

Structure


            <template>
              <button @click="count++">
                You clicked me {{ count }} times.
              </button>
            </template>

            <style>
              button {
                /* ... */
              }
            </style>

            <script>
              export default {
                name: 'counter-button',
                data() {
                  return {
                    count: 0
                  };
                }
              }
            </script>

Prop Features

Prop Casing

              <h3>{{ postTitle }}</h3>
            </template>

            <script>
              export default {
                name: 'blog-post',
                props: ['postTitle']
              }
            </script>
          

            <blog-post post-title="Another blog"></blog-post>

Prop Types


            <script>
              export default {
                props: ['title', 'likes', 'isPublished', 'commentIds', 'author']
              }
            </script>
          

            <script>
              export default {
                props: {
                  title: String,
                  likes: Number,
                  isPublished: Boolean,
                  commentIds: Array,
                  author: Object,
                  callback: Function,
                  custom: Custom
                }
              }
            </script>

Prop Validation


            <script>
              export default {
                props: {
                  propA: Number,
                  propB: [String, Number],
                  propC: {
                    type: String,
                    required: true
                  },
                  propD: {
                    type: Number,
                    default: 42
                  },
                  propE: {
                    type: Object,
                    default: function() {
                      returns { message: 'hello' };
                    }
                  },
                  propF: {
                    type: Array,
                    default: function() {
                      returns [3.41, 1729];
                    }
                  },
                  propG: {
                    validator: function(value) {
                      return ['success', 'warning', 'danger'].indexOf(value) !== -1;
                    }
                  }
                }
              }
            </script>

Static vs Dynamic Props

Static Props


            <template>
              <blog-post title="My journey with Vue"></blog-post>
            </template>

Dynamic Props


            <template>
              <blog-post :title="post.title"></blog-post>
            </template>

            <script>
              export default {
                name: 'blog-container',
                data() {
                  return {
                    post: {
                      title: 'Simplicity of Vue'
                    }
                  }
                }
              }
            </script>

Passing a Number


            <template>
              <blog-post :likes="42"></blog-post>
            </template>
          

            <template>
              <blog-post :likes="post.likes"></blog-post>
            </template>

Passing a Boolean


            <template>
              <blog-post is-published></blog-post>
            </template>
          

            <template>
              <blog-post :is-published="false"></blog-post>
            </template>
          

            <template>
              <blog-post :is-published="post.isPublished"></blog-post>
            </template>

Passing an Array


            <template>
              <blog-post :comment-ids="[234, 266, 273]"></blog-post>
            </template>
          

            <template>
              <blog-post :comment-ids="post.commentIds"></blog-post>
            </template>

Passing an Object


            <template>
              <blog-post :author="{ name: 'Veronica', company: 'Veridian Dynamics' }"></blog-post>
            </template>
          

            <template>
              <blog-post :author="post.author"></blog-post>
            </template>

Passing Properties of an Object


            <template>
              <blog-post v-bind="post"></blog-post>
            </template>

            <script>
              export default {
                name: 'blog-container',
                data() {
                  return {
                    post: {
                      id: 1,
                      title: 'My Journey with Vue'
                    }
                  }
                }
              }
            </script>
          

            <template>
              <blog-post :id="post.id" :title="post.title"></blog-post>
            </template>

Events

Example


            <template>
              <div :style="{ fontSize: fontSize + 'em' }">
                <blog-post :title="title"></blog-post>
              </div>
            </template>

            <script>
              export default {
                name: 'blog-container',
                data() {
                  return {
                    title: 'Casablanca is a timeless classic.',
                    fontSize: 1
                  }
                }
              }
            </script>
          

            <template>
              <div>
                <h3>{{ title }}</h3>

                <button>
                  Enlarge text
                </button>
              </div>
            </template>

            <script>
              export default {
                name: 'blog-post'
              }
            </script>

Solution


            <template>
              <div>
                <h3>{{ title }}</h3>

                <button @click="$emit('enlarge-text')">
                  Enlarge text
                </button>
              </div>
            </template>

            <script>
              export default {
                name: 'blog-post'
              }
            </script>
          

            <template>
              <div :style="{ fontSize: fontSize + 'em' }">
                <blog-post @enlarge-text="fontSize += 0.1" :title="title"></blog-post>
              </div>
            </template>

            <script>
              export default {
                name: 'blog-container',
                data() {
                  return {
                    title: 'Casablanca is a timeless classic.',
                    fontSize: 1
                  }
                }
              }
            </script>

Events with Custom Values


            <template>
              <div>
                <h3>{{ title }}</h3>

                <button @click="$emit('enlarge-text', 0.1)">
                  Enlarge text
                </button>
              </div>
            </template>

            <script>
              export default {
                name: 'blog-post'
              }
            </script>
          

            <template>
              <div :style="{ fontSize: fontSize + 'em' }">
                <blog-post @enlarge-text="fontSize += $event" :title="title"></blog-post>
              </div>
            </template>

            <script>
              export default {
                name: 'blog-container',
                data() {
                  return {
                    title: 'Casablanca is a timeless classic.',
                    fontSize: 1
                  }
                }
              }
            </script>
